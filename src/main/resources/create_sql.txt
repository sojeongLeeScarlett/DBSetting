

-- 커피전문점판매실적
DROP SCHEMA IF EXISTS coffeeprice;

-- 커피전문점판매실적
CREATE SCHEMA coffeeprice;



-- 제품명 
CREATE TABLE product (
	pdCode VARCHAR(50) NOT NULL COMMENT '제품코드', -- 제품코드
	pdName VARCHAR(50) NULL     COMMENT '제품명' -- 제품명
)
COMMENT '제품명 ';

-- 제품명 
ALTER TABLE product
	ADD CONSTRAINT PK_product -- 제품명  기본키
		PRIMARY KEY (
			pdCode -- 제품코드
		);

-- 제품상세
CREATE TABLE pdintro (
	pdCode      VARCHAR(50) NOT NULL COMMENT '제품코드', -- 제품코드
	pdName      VARCHAR(50) NULL     COMMENT '제품명', -- 제품명
	pdUnitprice INTEGER(8)  NULL     COMMENT '제품단가', -- 제품단가
	pdUnitsales INTEGER(8)  NULL     COMMENT '판매수량', -- 판매수량
	pdPermargin INTEGER(2)  NULL     COMMENT '마진율' -- 마진율
)
COMMENT '제품상세';

-- 제품상세
ALTER TABLE pdintro
	ADD CONSTRAINT PK_pdintro -- 제품상세 기본키
		PRIMARY KEY (
			pdCode -- 제품코드
		);


-- create and grant user
grant select, insert, update,delete on mysql.* to 'coffeeprice' identified by '1234';
